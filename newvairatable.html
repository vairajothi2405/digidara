CREATE TABLE student_certificates ( id INT AUTO_INCREMENT PRIMARY KEY, regno VARCHAR(50), name VARCHAR(100), course VARCHAR(100), department VARCHAR(100), provider VARCHAR(100), email VARCHAR(100), certificate_link TEXT, FOREIGN KEY (regno) REFERENCES students(regno) ); CREATE TABLE students ( id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(100), regno VARCHAR(50) UNIQUE, email VARCHAR(100) UNIQUE, course VARCHAR(100), department VARCHAR(100) ); CREATE TABLE admin_users ( id INT AUTO_INCREMENT PRIMARY KEY, username VARCHAR(50) UNIQUE, password VARCHAR(100) -- You can hash passwords later for better security );
